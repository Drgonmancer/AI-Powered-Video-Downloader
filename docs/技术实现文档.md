```python
    def __init__(self):
        self.active: Set[WebSocket] = set()

    async def connect(self, ws: WebSocket):
        await ws.accept()
        self.active.add(ws)

    def disconnect(self, ws: WebSocket):
        self.active.discard(ws)

    async def broadcast(self, message: dict):
        for ws in list(self.active):
            try:
                await ws.send_json(message)
            except:
                self.active.discard(ws)
```

```python
async def parse_via_iesdouyin_share_page(aweme_id: str) -> dict:
    """
    核心实现原理：
    1. 访问 https://www.iesdouyin.com/share/video/{aweme_id}/
    2. 使用移动端UA模拟手机浏览器
    3. 从HTML中提取 window._ROUTER_DATA JSON
    4. 解析JSON获取完整视频信息
    5. 替换 playwm → play 获取无水印URL
    """

    share_url = f'https://www.iesdouyin.com/share/video/{aweme_id}/'

    mobile_headers = {
        'User-Agent': (
            'Mozilla/5.0 (iPhone; CPU iPhone OS 16_0 like Mac OS X) '
            'AppleWebKit/605.1.15 (KHTML, like Gecko) Version/16.0 '
            'Mobile/15E148 Safari/604.1'
        ),
        'Accept-Language': 'zh-CN,zh;q=0.9,en;q=0.8',
        'Referer': 'https://www.douyin.com/',
    }

    async with httpx.AsyncClient(
        timeout=20,
        follow_redirects=True,
        headers=mobile_headers,
        verify=False
    ) as client:
        resp = await client.get(share_url)
        html = resp.text

        # 提取 _ROUTER_DATA JSON

        start_idx = html.find(marker)

        # 解析嵌套JSON（处理转义字符）
        payload = extract_json_from_html(html, start_idx)
        router_data = json.loads(payload)

        # 提取视频信息
        loader_data = router_data.get('loaderData', {})
        for node_key, node_value in loader_data.items():
            video_info_res = node_value.get('videoInfoRes', {})
            item_list = video_info_res.get('item_list', [])
```

## 三、downloader.py — yt-dlp 封装层（通用）

### 3.1 设计思路

yt-dlp 有两种使用方式：
- **命令行模式**：`subprocess` 调用 `yt-dlp` CLI → 进度解析麻烦，但进程好控制（pause/resume靠kill）
- **Python API模式**：直接 `import yt_dlp` → 进度回调方便，但暂停/恢复需要特殊处理

**决策：采用 Python API 模式**
- 进度通过 `progress_hooks` 回调获取，实时性好
- 暂停：在回调中设置标志位，让下载线程主动退出
- 恢复：重新创建YoutubeDL实例重新下载（yt-dlp自动检测.part文件续传）

### 3.2 解析功能实现

```python
# services/downloader.py

import yt_dlp
from urllib.parse import urlparse

def parse_video(url: str) -> dict:
    """
    解析视频URL，返回标准化信息
    不下载任何内容，仅提取元数据和格式列表
    """
    ydl_opts = {
        'quiet': True,
        'no_warnings': True,
        'extract_flat': False,     # 完整提取播放列表
    }

    with yt_dlp.YoutubeDL(ydl_opts) as ydl:
        info = ydl.extract_info(url, download=False)
        return normalize_info(info)


def normalize_info(raw_info: dict) -> dict:
    """
    将yt-dlp原始输出转换为前端需要的格式
    """
    formats = []
    if raw_info.get('formats'):
        for f in raw_info['formats']:
            if f.get('vcodec') != 'none' or f.get('acodec') != 'none':
                formats.append({
                    'format_id': f['format_id'],
                    'ext': f.get('ext', ''),
                    'resolution': f.get('resolution', f.get('format_note', '')),
                    'filesize': f.get('filesize') or f.get('filesize_approx', 0),
                    'vcodec': f.get('vcodec', ''),
                    'acodec': f.get('acodec', ''),
                    'tbr': f.get('tbr', 0),
                })

    best = pick_best_format(formats)

    return {
        'title': raw_info.get('title', ''),
        'description': raw_info.get('description', '')[:500],
        'thumbnail': raw_info.get('thumbnail', ''),
        'duration': raw_info.get('duration', 0),
        'platform': get_platform(raw_info.get('webpage_url', raw_info.get('url', ''))),
        'uploader': raw_info.get('uploader', ''),
        'formats': formats,
        'is_playlist': '_type' in raw_info and raw_info['_type'] == 'playlist',
        'playlist_count': raw_info.get('playlist_count') if raw_info.get('_type') == 'playlist' else None,
        'entry_count': raw_info.get('entry_count', len(raw_info.get('entries', []))),
        'entries': [e['url'] for e in raw_info.get('entries', [])] if raw_info.get('entries') else [],
        'best_format': best,
    }


def pick_best_format(formats: list[dict]) -> dict:
    """
    智能选择最佳格式
    策略：
    1. 找MP4容器且包含视频+音频的合并流
    2. 若无合并流，找最高分辨率视频流 + 最高音质音频流
    3. 音频-only场景选最高音质
    """
    video_formats = [f for f in formats if f['vcodec'] not in ('none', '')]
    audio_formats = [f for f in formats if f['acodec'] not in ('none', '')]

    if not video_formats and audio_formats:
        best_audio = max(audio_formats, key=lambda x: x.get('tbr', 0))
        return {'format_id': best_audio['format_id'], 'resolution': 'audio only',
                'ext': best_audio['ext'], 'note': '最佳音质'}

    merged_mp4 = [f for f in formats
                  if f['vcodec'] not in ('none', '')
                  and f['acodec'] not in ('none', '')
                  and f['ext'] == 'mp4']
    if merged_mp4:
        best = max(merged_mp4, key=lambda x: x.get('tbr', 0))
        return {'format_id': best['format_id'], 'resolution': best['resolution'],
                'ext': best['ext'], 'note': '最佳画质合并流'}

    if video_formats and audio_formats:
        best_v = max(video_formats, key=lambda x: x.get('tbr', 0))
        best_a = max(audio_formats, key=lambda x: x.get('tbr', 0))
        return {'format_id': f"{best_v['format_id']}+{best_a['format_id']}",
                'resolution': best_v['resolution'],
                'ext': 'mp4', 'note': '视频+音频合并（需合并）'}

    return {'format_id': '', 'resolution': '', 'ext': '', 'note': '无可选格式'}


def get_platform(url: str) -> str:
    domain = urlparse(url).netloc.lower()
    platform_map = {
        'youtube.com': 'youtube', 'youtu.be': 'youtube',
        'bilibili.com': 'bilibili', 'b23.tv': 'bilibili',
        'douyin.com': 'douyin', 'iesdouyin.com': 'douyin',
        'kuaishou.com': 'kuaishou',
        'twitter.com': 'twitter', 'x.com': 'twitter',
        'instagram.com': 'instagram',
        'tiktok.com': 'tiktok',
    }
    for domain_key, name in platform_map.items():
        if domain_key in domain:
            return name
    return 'unknown'
```

### 3.3 下载功能实现（异步+进度推送）

```python
# services/downloader.py 续

import threading
import asyncio


class DownloadEngine:
    """
    下载引擎：封装yt-dlp的下载操作
    每个Task对应一个DownloadEngine实例
    """

    def __init__(self, task: 'Task', config: dict,
                 on_progress=None, on_complete=None, on_error=None):
        self.task = task
        self.config = config
        self.on_progress = on_progress
        self.on_complete = on_complete
        self.on_error = on_error
        self._cancelled = False
        self._ydl = None

    def build_ydl_opts(self) -> dict:
        """根据task和config构建yt-dlp选项"""
        opts = {
            'quiet': True,
            'no_warnings': True,
            'outtmpl': os.path.join(
                self.config['download_path'],
                '%(title)s.%(ext)s'
            ),
            'merge_output_format': self.task.output_format,
            'progress_hooks': [self._progress_hook],
            'postprocessors': [],  # 后期可加缩略图等后处理
        }

        if self.config.get('speed_limit', 0) > 0:
            opts['ratelimit'] = self.config['speed_limit']

        if self.task.format_id:
            opts['format'] = self.task.format_id

        return opts

    def _progress_hook(self, d: dict):
        """yt-dlp进度回调 —— 这是整个系统的核心事件源"""
        try:
            if d['status'] == 'downloading':
                total = d.get('total_bytes') or d.get('total_bytes_estimate', 1)
                downloaded = d.get('downloaded_bytes', 0)
                self.task.progress = (downloaded / total) * 100
                self.task.speed = d.get('_speed_str', '')
                self.task.eta = d.get('_eta_str', '')

                if self.on_progress:
                    self.on_progress(self.task.to_dict())

            elif d['status'] == 'finished':
                self.task.progress = 100
                self.task.status = TaskStatus.MERGING
                if self.on_progress:
                    self.on_progress(self.task.to_dict())

            elif d['status'] == 'error':
                self.task.error_message = d.get('error', '未知错误')
                self.task.status = TaskStatus.FAILED
                if self.on_error:
                    self.on_error(self.task.to_dict())
        except Exception as e:
            print(f"Progress hook error: {e}")

    def run_in_thread(self):
        """在线程中执行下载（阻塞式），由asyncio调度"""
        try:
            self.task.status = TaskStatus.DOWNLOADING
            self.task.started_at = time.time()
            opts = self.build_ydl_opts()

            with yt_dlp.YoutubeDL(opts) as self._ydl:
                info = self._ydl.extract_info(self.task.url, download=True)
                if info:
                    self.task.title = info.get('title', self.task.title)
                    expected_filename = self._ydl.prepare_filename(info)
                    self.task.filepath = expected_filename
                    if os.path.exists(self.task.filepath):
                        self.task.filesize = os.path.getsize(self.task.filepath)

            self.task.status = TaskStatus.COMPLETED
            self.task.completed_at = time.time()
            if self.on_complete:
                self.on_complete(self.task.to_dict())

        except Exception as e:
            self.task.error_message = str(e)
            self.task.status = TaskStatus.FAILED
            if self.on_error:
                self.on_error(self.task.to_dict())

    def cancel(self):
        """取消下载"""
        self._cancelled = True
        if self._ydl:
            try:
                self._ydl.params['abort'] = True
            except:
                pass
```

### 3.4 在 main.py 中集成下载引擎

```python
# main.py 中的关键部分

import asyncio
import threading

async def start_download(task: Task):
    """
    异步启动下载任务
    使用信号量控制并发数
    使用线程池执行阻塞式的yt-dlp下载
    """
    async with semaphore:
        engine = DownloadEngine(
            task=task,
            config=config,
            on_progress=_on_task_update,
            on_complete=_on_task_update,
            on_error=_on_task_update,
        )
        tasks[task.id].status = TaskStatus.DOWNLOADING

        loop = asyncio.get_event_loop()

        def _run():
            engine.run_in_thread()

        await loop.run_in_executor(None, _run)


def _on_task_update(task_data: dict):
    """任务状态变更回调 → 广播给所有WebSocket客户端"""
    tasks[task_data['id']].__dict__.update({
        k: v for k, v in task_data.items() if k in tasks[task_data['id']].to_dict()
    })
    asyncio.create_task(ws_manager.broadcast(task_data))
```

### 3.5 暂停/恢复机制

```python
# 暂停：终止线程 + 保留.part文件（yt-dlp自动保留）
@app.post("/api/download/{task_id}/pause")
async def pause_download(task_id: str):
    task = tasks.get(task_id)
    if not task or task.status != TaskStatus.DOWNLOADING:
        return {"error": "任务不可暂停"}

    task.status = TaskStatus.PAUSED
    # yt-dlp没有真正的pause API，我们通过标记取消来实现
    # .part文件会保留，下次resume时自动续传
    await ws_manager.broadcast(task.to_dict())
    return {"ok": True}

# 恢复：重新创建下载引擎，从.part文件续传
@app.post("/api/download/{task_id}/resume")
async def resume_download(task_id: str):
    task = tasks.get(task_id)
    if not task or task.status != TaskStatus.PAUSED:
        return {"error": "任务不可恢复"}

    task.error_message = ""  # 清除错误
    asyncio.create_task(start_download(task))  # 重新加入队列
    await ws_manager.broadcast(task.to_dict())
    return {"ok": True}
```

### 4.1 实现

```python
# services/converter.py
import subprocess
import json

def convert_file(input_path: str, output_format: str,
                 quality: str = "medium") -> dict:
    """
    使用ffmpeg进行格式转换
    quality: high(原质量) / medium(CRF 23) / low(CRF 30)
    """
    input_ext = input_path.rsplit('.', 1)[-1]
    base_name = input_path.rsplit('.', 1)[0]
    output_path = f"{base_name}.{output_format}"

    quality_map = {
        'high': ['-crf', '18'],
        'medium': ['-crf', '23'],
        'low': ['-crf', '30'],
    }

    cmd = ['ffmpeg', '-y', '-i', input_path]

    if output_format in ('mp4', 'webm', 'mkv'):
        cmd += quality_map.get(quality, quality_map['medium'])
        cmd += ['-c:a', 'aac', '-b:a', '192k']
    elif output_format in ('mp3',):
        cmd += ['-vn', '-ab', '192k', '-ar', '44100']
    elif output_format in ('aac',):
        cmd += ['-vn', '-c:a', 'aac', '-b:a', '256k']
    elif output_format in ('flac',):
        cmd += ['-vn', '-c:a', 'flac']
    elif output_format in ('wav',):
        cmd += ['-vn', '-c:a', 'pcm_s16le']

    cmd.append(output_path)

    process = subprocess.Popen(
        cmd,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        text=True
    )

    _, stderr = process.communicate(timeout=3600)  # 1小时超时

    if process.returncode == 0:
        file_size = os.path.getsize(output_path) if os.path.exists(output_path) else 0
        return {
            "success": True,
            "output_path": output_path,
            "file_size": file_size,
        }
    else:
        return {
            "success": False,
            "error": stderr[-500:] if stderr else "转换失败",
        }
```
